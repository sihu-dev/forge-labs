### 2.3 AI 모델 아키텍처

#### RAG (Retrieval-Augmented Generation) 방식

```python
# 의사코드
class QuoteGenerationEngine:
    def __init__(self):
        self.vector_db = ChromaDB()  # 벡터 DB (레퍼런스 임베딩)
        self.llm = ClaudeOpus4()     # LLM (견적 생성)

    def generate_quote(self, rfq: RFQ) -> Quote:
        # 1. RFQ를 임베딩 벡터로 변환
        rfq_embedding = self.embed(rfq)

        # 2. 유사 프로젝트 검색 (TOP 5)
        similar_projects = self.vector_db.similarity_search(
            rfq_embedding,
            k=5,
            filters={
                "product_type": rfq.product_type,
                "diameter_range": [rfq.diameter * 0.8, rfq.diameter * 1.2],
                "year": [2020, 2024]  # 최근 5년 데이터만
            }
        )

        # 3. 컨텍스트 구성
        context = {
            "similar_projects": similar_projects,
            "product_catalog": self.get_products(rfq.product_type),
            "pricing_rules": self.get_pricing_rules(),
            "supply_chain_status": self.get_supply_chain_status()
        }

        # 4. LLM으로 견적서 생성
        quote = self.llm.generate(
            prompt=f"고객 요구사항: {rfq}\n유사 프로젝트: {context}",
            temperature=0.2  # 낮은 temperature (정확성 중시)
        )

        return quote
```

### 3.3 AI 모델 아키텍처

#### 하이브리드 방식 (템플릿 + RAG)

```python
class TechnicalSpecGenerator:
    def __init__(self):
        self.template_db = TemplateDatabase()
        self.vector_db = ChromaDB()
        self.llm = ClaudeOpus4()
        self.cad_engine = AutoCAD_API()  # 선택 사항

    def generate_spec(self, requirements: Requirements) -> Document:
        # 1. 적절한 템플릿 선택
        template = self.template_db.get_template(
            product_type=requirements.product_type,
            language="ko"
        )

        # 2. 유사 사양서 검색
        similar_specs = self.vector_db.similarity_search(
            query=requirements.description,
            k=3,
            filters={"product_type": requirements.product_type}
        )

        # 3. 제품 스펙 조회
        product_specs = self.get_product_specs(requirements.model)

        # 4. LLM으로 사양서 생성
        sections = {}
        for section in template.sections:
            sections[section.name] = self.llm.generate(
                prompt=f"""
                템플릿: {section.template}
                제품 스펙: {product_specs}
                참고 사양서: {similar_specs}
                고객 요구사항: {requirements}
```

#### 하이브리드 방식 (Rule-Based + ML + RAG)

```python
class FaultDiagnosticEngine:
    def __init__(self):
        self.rule_engine = RuleEngine()  # 에러 코드 → 원인 매핑
        self.ml_model = RandomForestClassifier()  # 계측 데이터 패턴 분류
        self.vector_db = ChromaDB()  # A/S 이력 검색
        self.llm = ClaudeOpus4()  # 진단 리포트 생성

    def diagnose(self, issue: Issue) -> Diagnosis:
        # 1. 룰 기반 진단 (에러 코드)
        if issue.error_code:
            rule_diagnosis = self.rule_engine.diagnose(issue.error_code)
            candidates = rule_diagnosis.possible_causes  # 예: ["전극 오염", "케이블 단선"]
        else:
            candidates = []

        # 2. ML 기반 진단 (계측 데이터 패턴)
        if issue.time_series_data:
            features = self.extract_features(issue.time_series_data)
            ml_diagnosis = self.ml_model.predict_proba(features)
            candidates.extend(ml_diagnosis.top_k(k=3))

        # 3. 유사 케이스 검색 (RAG)
        similar_cases = self.vector_db.similarity_search(
            query=f"{issue.product} {issue.symptoms} {issue.error_code}",
            k=5,
            filters={"resolution_status": "resolved"}
        )

        # 4. 확률 계산 (앙상블)
        probabilities = self.ensemble_probabilities(
            rule_diagnosis, ml_diagnosis, similar_cases
        )

        # 5. LLM으로 진단 리포트 생성
        diagnosis_report = self.llm.generate(
            prompt=f"""
            제품: {issue.product}
            증상: {issue.symptoms}
            에러 코드: {issue.error_code}
            계측 데이터 패턴: {ml_diagnosis}
            유사 A/S 사례: {similar_cases}
            추정 원인 (확률): {probabilities}

            위 정보를 바탕으로 진단 리포트를 작성하세요:
            1. 추정 원인 (확률 높은 순, 최대 3개)
            2. 조치 방안 (단계별)
            3. 예비부품 목록 (모델명, 가격 포함)
            """,
            temperature=0.2
        )

        return diagnosis_report

    def extract_features(self, time_series: TimeSeries) -> Features:
        """시계열 데이터에서 특징 추출"""
        return {
            "trend": time_series.calculate_trend(),  # 상승/하락/변동 없음
            "volatility": time_series.std(),
            "sudden_change": time_series.detect_sudden_change(),
            "periodicity": time_series.detect_periodicity(),
        }
```
